\section{La construcción espacial}
\begin{frame}{La idea de Vietoris}
Sea $S\in \Top$ y $\mathcal{H}S$ una familia de subconjuntos de $S$
\[
(\text{cerrados, compactos, compactos saturados,} \dots). 
\]

Dotamos de una topología a $\mathcal{H}S$ por medio de condiciones de pertenencia y de intersección con abiertos de $S$.
\begin{itemize}
\item \textbf{[``Hit'' o $\Diamond(U)$]:} $H\in \mathcal{H}S$ intersecta a un abierto $U$ de $S$.
\item \textbf{[``Miss'' o $\Box(U)$]:} $H\in \mathcal{H}S$ está contenido en un abierto $U$ de $S$.
\end{itemize}

\[
\langle U_1,\dots,U_n\rangle
=
\left\{
K\in\mathcal{H}(S)
\mid
K\subseteq \bigcup_{i=1}^n U_i
\text{ y }
K\cap U_i\neq\varnothing
\ \forall i
\right\}
\]
\end{frame}

\begin{frame}{Aspectos topológicos}
\begin{itemize}
    \item Sea $S\in \Top$ y $p, q\in S$, entonces 
    \[
    q \sqsubseteq p \iff q\in \overline{\{p\}}
    \]
    es el orden de especialización en $S$.
    \item Las familias de subconjuntos de $S$
    \[
    \mathcal{O}S, \quad \mathcal{C}S, \quad \mathcal{K}S, \quad \mathcal{U}S, \quad \mathcal{Q}S
    \]
    abiertos, cerrados, compactos, saturados y compactos saturados.

    \item Para $E\subseteq S$, entonces 
    \[
        E^\circ, \quad \overline{E}, \quad \uparrow E
    \]
    denotan interior, cerradura y saturación de $E$.
    \end{itemize}
\end{frame}

\begin{frame}{La V-modificación de un espacio}
Sea $S\in \Top$ y $\mathcal{K}S$ . Para $U\in \mathcal{O}S$, definimos
\[
    \Box (U) = \{K\in \mathcal{K}S\mid K\subseteq U\}, \quad \Diamond (U) = \{K\in \mathcal{K}S\mid K\cap U\neq \emptyset\}.
\]
Entonces, la \textbf{modificación de Vietoris} de $S$ es el espacio topológico $VS$ con conjunto subyacente $\mathcal{K}S$ y topología generada por la subbase
\[
    \{\Box U: U\in \mathcal{O}S\}\cup \{\Diamond U: U\in \mathcal{O}S\}.
\]

\end{frame}
