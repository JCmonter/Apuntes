\begin{frame}[fragile]{Espacio de parches}
    Consideremos $S\in \Top$. Denotamos por $^pS=(S, \mathcal{O}^pS)$ al \textbf{espacio de parches}\footnote{El espacio de parches fue introducido en \cite{hochster1969prime}} de $S$, donde $\mathcal{O}^pS$ está generado por
    \[
    \mbox{pbase}=\{U\cap Q'\mid U\in \mathcal{O}S, Q\in \mathcal{Q}S\}
    \]
   \[\begin{tikzcd}
	{\mathcal{O}S} & {\mathcal{O}^pS}
	\arrow[from=1-1, to=1-2]
\end{tikzcd}\]

\onslide<2->{
\begin{block}{Definición \citep{sexton2006point}:}
$S$ es \textbf{empaquetado} si todo subconjunto compacto (saturado) es cerrado.
\end{block}
\[
S \mbox{ es empaquetado}\iff\, ^pS=S\quad \mbox{ y }\quad T_2\Rightarrow \mbox{empaquetado}\Rightarrow T_1
\]}
\end{frame}

\begin{frame}[fragile]{Marco de parches}
Consideremos $A\in \Frm$. Denotamos por $PA$ al \textbf{marco de parches}\footnote{Información sobre el marco de parches puede ser consultada en \cite{sexton2006point, sexton2006ordinal}} de $A$, donde $PA$ está generado por
\[
\mbox{Pbase}=\{u_a\wedge v_F\mid a\in A, F\in A^\wedge\}
\]

\[\begin{tikzcd}
	A & PA
	\arrow[from=1-1, to=1-2]
\end{tikzcd}\]

\onslide<2->{
\begin{block}{Definición \citep{sexton2006point}:}
    $A$ es \textbf{parche trivial} si y solo si $A\cong PA$.
\end{block}
      \[
        ¿ \; ?\Rightarrow \mbox{ parche trivial }\Rightarrow T_1
      \]
}
\end{frame}

\begin{frame}{Marcos eficientes}
Sean $A\in \Frm$ y $F\in A^\wedge$. Consideramos
\[ 
f_F=\dot{\bigvee}\{v_a\mid a\in F\}\in PrA.
\]
\begin{block}{Definición \citep{sexton2006point}:}
Para $\alpha\in \Ord$ decimos que:
\begin{enumerate}
    \item<2-> $F$ es $\alpha$\textbf{-eficiente} si para $x\in F$, $d\vee x=1$, donde $d=f_F^\alpha(0)$.
    \item<3-> $A$ es $\alpha$\textbf{-eficiente} si cada $F\in A^\wedge$ es $\alpha$-eficiente.
    \item<4-> $A$ es \textbf{eficiente} si es $\alpha$-eficiente para algún $\alpha\in \Ord$.
\end{enumerate}
\end{block}

\onslide<5->{\begin{block}{Proposición \citep{sexton2003point}}
\[
    A \mbox{ es eficiente}\quad \iff\quad A \mbox{ es parche trivial}.
\]
\end{block}}
\end{frame}

\begin{frame}{Para qué sirve la eficiciencia}
Para $S\in \Top$
  \[
\begin{split}
T_2\Rightarrow 1\text{-eficiente}\Rightarrow\dots & \Rightarrow \alpha\text{-eficiente}\Rightarrow\alpha+1\text{-eficiente}\\
& \Rightarrow \dots \Rightarrow T_1.
\end{split}
\]
\begin{itemize}
\item Clasifica espacios a través de la eficiencia de su topología.
\end{itemize}

\onslide<2->{Para $A\in \Frm$ y $F\in A^\wedge$
\begin{itemize}
\item Por construcción
\[
u_e=v_F, \quad \text{donde }e=v_F(0) \text{ y cada } F\in A^\wedge.
\]
\item Cada filtro abierto proporciona un cociente cerrado compacto. 
\item Refina la Pbase.
\end{itemize}}
\end{frame}

\begin{frame}{Propiedades de los marcos eficientes}
  Para $S\in \Top$
  \begin{itemize}
    \item $S$ es regular $\iff$ $\mathcal{O}S$ es 0-eficiente.
    \item $S$ es $T_2$ $\iff$ $\mathcal{O}S$ es 1-eficiente.
    \item $S$ es \alert<3->{apilado} y empaquetado $\iff$ $\mathcal{O}S$ es eficiente.
  \end{itemize}

  \onslide<2->{Para $A\in \Frm$
  \begin{itemize}
    \item $\mathbf{(reg)}\Rightarrow \mathbf{(aju)}\Rightarrow \text{eficiencia}$.
    \item $\mathbf{(fH)}\Rightarrow \text{eficiencia}$.
    \item $\text{Eficiencia}\Rightarrow T_1$.
    \item Son cerrados bajo cocientes y coproductos.
  \end{itemize}}
\end{frame}

\begin{frame}{Nociones de apilamiento}
Consideremos $S\in\Top$, $Q\in \mathcal{Q}S$ y $\nabla\in \mathcal{O}S^\wedge$. Al par $(\nabla, Q)$ lo llamamos par HM.
Adicionalmente, para $M=\mathcal{O}S\setminus \nabla$

\begin{equation}\label{Intervalo}
v_\nabla\leq [Q']\leq w_\nabla=[M']
\end{equation}

\begin{block}{Definición \citep{sexton2006ordinal}:}
  Sean $S\in \Top$, $F\in A^\wedge$ y $(F, Q)$ un par HM.
\begin{enumerate}
  \item Un conjunto $Q\in \mathcal{Q}S$ es apilado si $\overline{Q}=\partial_\nabla^\infty(S)$ y es fuertemente apilado si $v_\nabla=[Q']$.
  \item El espacio $S$ es apilado o fuertemente apilado si cada $Q\in \mathcal{Q}S$ es apilado o fuertemente apilado, respectivamente.
\end{enumerate}
\end{block}

Dado que $v_\nabla, [Q'], w_\nabla\in N\mathcal{O}S$, sus levantamientos están en $NA$.
\end{frame}

\begin{frame}
\begin{block}{Proposición:}
Sean $A\in \Frm$, $S=\pt A$ y $(F, Q, \nabla)$ una terna HM. Para $v_\nabla, [Q']$ y $w_\nabla$ como antes, las siguientes son ciertas:
\begin{enumerate}
\item $\mathcal{U}_*\circ v_\nabla\circ \mathcal{U}, \mathcal{U}_*\circ [Q']\circ \mathcal{U}, \mathcal{U}_*\circ w_\nabla\circ \mathcal{U}\in [v_F, w_F]$
\item $\mathcal{U}_*\circ w_\nabla\circ \mathcal{U}=w_F$.
\end{enumerate}
\end{block}

\begin{proof}
\begin{enumerate}
\item Sabemos que $\mho\colon [v_\nabla, w_\nabla]\to [v_F, w_F]$ y $v_F$, $w_F$ son el menor y mayor elemento de su bloque, respectivamente.
\item Por propiedades del adjunto derecho, verificamos que $\mathcal{U}\circ w_F\subseteq  w_\nabla\circ \mathcal{U}$.
\end{enumerate}
\end{proof}
\end{frame}


\begin{frame}{Marcos apilados y fuertemente apilados}
\begin{block}{Definición:}
Sean $A\in \Frm$, $S=\pt A$ y $(F,Q)$ un par HM. Decimos que $A$ es:
\begin{enumerate}
  \item apilado si $(\mathcal{U}_*\circ [Q']\circ\mathcal{U})(0)=v_F(0)$.
  \item fuertemente apilado si $\mathcal{U}_*\circ [Q']\circ \mathcal{U}=v_F$.
\end{enumerate}
\end{block}

\begin{block}{Proposición:}
Las nociones de fuertemente apilado y apilado son conservativas.
\end{block}

\begin{proof}
  Verificamos que 
\[
S \text{ apilado (f. apilado)}\iff \mathcal{O}S \text{ apilado (f. apilado)}.
\]
\end{proof}
\end{frame}

\begin{frame}{Apilamiento con otras propiedades}
En espacios, si $S\in \Top$ es $T_1$ y fuertemente apilado, entonces $v_\nabla=w_\nabla$

\begin{block}{Proposición:}
Si $A\in \Frm$ es $T_1$ y fuertemente apilado, entonces $v_F=w_F$.
\end{block}

\begin{proof}
Se sigue del hecho de que $T_1$  es una propiedad hereditaría y de que $T_1$ y fuertemente apilado son conservativas. 
\end{proof}
\end{frame}

\begin{frame}
\begin{block}{Teorema \citep[6.2]{sexton2006ordinal}:}
  Para $S\in \Top$, las siguientes implicaciones se cumplen:
    \[
      \text{eficiente}\Rightarrow \text{f. apilado}\Rightarrow \text{apilado}\quad T_1+\text{apilado}\Rightarrow \text{sobrio}.
    \]
\end{block}

\begin{block}{Teorema \citep[7.4]{sexton2006ordinal}:}
Un espacio $S$ es $T_1$ y apilado si y solo si $S$ es sobrio y sus $V$-modificaciones $\mathcal{Q}S$ y $\mathcal{V}S$ son canónicamente homeomorfas.
\end{block}

\begin{block}{Teorema \citep[7.11]{simmons2004vietoris}:}
Sea $S$ un espacio sobrio. $S$ es $T_1$ y apilado si y solo si $\dagger Q$ tiene solo un elemento para cada $Q\in \mathcal{Q}S$, en otras palabras, \eqref{Intervalo} evaluado en $\emptyset$ colapsa.
\end{block}
\end{frame}

\begin{frame}
\begin{block}{Proposición:}
Sean $A\in \Frm$, $S=\pt A$ y $(F, Q)$ cualquier par HM. Si $A$ es $T_1$ y apilado, las siguientes afirmaciones se cumplen:
\begin{enumerate}
\item $v_F(0)=w_F(0)$
\item Para cualquier para $(F_p, \uparrow p)$, $v_{F_p}=w_{F_p}=w_p$.
\item Para todo $p\in S$, p es máximo en $S$.
\end{enumerate}
\end{block}

\begin{proof}
\begin{enumerate}
\item $v_F(0)=(\mathcal{U}_*\circ [Q']\circ \mathcal{U})(0)$ y $\mathcal{U}_*\circ[Q']\circ\mathcal{U}=w_F$.
\item $\uparrow p=\{p\}$ y $A_{F_p}=\{p,1\}$. Por lo tanto, $v_{F_p}=w_{F_p}=w_p$. 
\item Por 1), $v_{F_p}(0)=w_{F_p}(0)$ y considerando $p\leq q\in \pt A$, vemos que $p=q$.
\end{enumerate}
\end{proof}
\end{frame}

\begin{frame}
Con respecto a la eficiencia, en espacios se cumple que
\begin{equation}\label{eficiencia}
\mathcal{O}S\text{ eficiente }\iff S \text{ apilado y empaquetado}
\end{equation}

\begin{block}{Proposición}
Si $\mathcal{O}S$ es eficiente, entonces $S$ es fuertemente apilado
\end{block}

\begin{proof}
Por \eqref{eficiencia}, $S$ es apilado y empaquetado. Por empaquetado, 
\[
Q\in \mathcal{C}S \quad\text{y}\quad [Q']=u_{Q'}.
\]
Por apilado,
\[
\overline{Q}=\partial_\nabla^\infty(S)\Rightarrow Q'=v_\nabla(\emptyset).
\]
Por eficiencia, $v_\nabla=u_{Q'}=[Q']$.
\end{proof}
\end{frame}

\begin{frame}[fragile]
  Si consideramos $(F. Q, \nabla)$ una terna HM, entonces
  \begin{equation}\label{laxo}
  \begin{tikzcd}
	A && A & \\
	&&& {\text{ y }\quad\mathcal{U}\circ v_F\leq v_\nabla\circ \mathcal{U}.} \\
	{\mathcal{O}S} && {\mathcal{O}S}
	\arrow["{v_F}", from=1-1, to=1-3]
	\arrow["{\mathcal{U}}"', from=1-1, to=3-1]
	\arrow["{\mathcal{U}}", from=1-3, to=3-3]
	\arrow["{v_\nabla}"', from=3-1, to=3-3]
\end{tikzcd}\end{equation}


\begin{block}{Proposición:}
Con los datos anteriores, las siguientes afirmaciones se cumplen:
\begin{enumerate}
  \item Si $A$ es fuertemente apilado, entonces \eqref{laxo} conmuta.
  \item Si $A$ es eficiente y \eqref{laxo} conmuta, entonces $A$ es fuertemente apilado.
\end{enumerate}
\end{block}
\end{frame}

\begin{frame}
\begin{proof}
\begin{enumerate}
\item Se sigue de la definición de fuertemente apilado y de la propiedad del adjunto derecho de $\mathcal{U}$.
\item Por la conmutatividad se cumple que
\[
\mathcal{U}\circ v_F\leq v_\nabla\circ \mathcal{U}\quad\text{y}\quad v_\nabla\circ \mathcal{U}\leq \mathcal{U}\circ v_F.
\]
Por la eficiencia,
\[
v_\nabla=[Q'].
\]
Sustituyendo y aplicando propiedades del adjunto derecho, obtenemos
\[
\mathcal{U}_*\circ [Q']\circ \mathcal{U}\leq v_F\quad\text{y}\quad v_F\leq \mathcal{U}_*\circ [Q']\circ \mathcal{U}.
\]
\end{enumerate}
\end{proof}
\end{frame}

\begin{frame}
\begin{block}{Proposición:}
Para $A\in \Frm$ y $(F, Q)$ cualquier par HM, se cumplen las siguientes afirmaciones:
\begin{enumerate}
\item Si $A$ es $T_1$ y apilado, entonces $v_F(0)=w_F(0)$.
\item Si $v_F(0)=w_F(0)$, entonces $A$ es apilado.
\item Si $v_F(0)=w_F(0)$ y el conjunto $Y=\{q\in \pt A\mid a\leq q\}$ es distinto de vacío para todo $a\in A$, entonces $A$ es $T_1$.
\end{enumerate}
\end{block}

\begin{proof}

\begin{description}

\]

\end{proof}

\begin{frame}{La V-modificación de un marco}
\begin{block}{Definición \citep{JohnstoneVietorisMonad}:}
Sea $\Vrm$ la categoría de $V$-marcos. 
\[
\Vrm=
\left\{
\begin{array}{ll}
\text{Objetos:} & \text{ marcos} 
\\[1mm]
\text{Morfismos:} & f_\Box\colon A\to B, \; f^\Diamond\colon A\to B \text{ tales que}
\end{array}
\right.
\]
\begin{description}
\item[($\wedge$)] $f_\Box(x\wedge y)=f_\Box(x)\wedge f_\Box(y)$ y $f_\Box(1)=1$.
\item[(c)] $f_\Box(\bigvee X)=\bigvee\{f_\Box(x)\mid x\in X\}$.
\item[($\bigvee$)] $f^\Diamond(\bigvee Y)=\bigvee\{f^\Diamond(y)\mid y\in Y\}$ y $f^\Diamond(0)=0$.
\item[(q)] $f^\Diamond(x)\wedge f_\Box(y)\leq f^\Diamond(x\wedge y)$.
\item[(s)] $f_\Box(x\vee y)\leq f_\Box(x)\vee f^\Diamond(y)$
\end{description}
para cada $x, y\in A$, $X, Y\subseteq A$, con $X$ dirigido.
\end{block}
\end{frame}

\begin{frame}[fragile]
\[
\begin{tikzcd}
\Vrm \arrow[rr, "V(-)", bend left] &  & \Frm \arrow[ll, "\iota (-)", bend left] &  & {\Frm(VA, B)\cong \Vrm(A, B)}
\end{tikzcd}.
\]
Llamamos a $VA$ el marco de Vietoris\footnote{Información sobre esta construcción puede ser consultada en \cite{JohnstoneVietorisMonad, johnstone2020vietoris, simmons2004vietoris}} asociado a $A$.
En particular,
\[
  \pt(VA)=\Frm(VA, 2)\cong \Vrm(A, 2).
\]
Para todo $p\in \pt VA$, existe 
\[
  (\Diamond_p,\Box_p)\colon A\to 2
\]
tal que $(\Diamond_p,\Box_p)\in \Vrm(A, 2)$.
\end{frame}

\begin{frame}
\begin{block}{Definición \citep{JohnstoneVietorisMonad}:}
Sea $A\in \Frm$ y $p\in \pt VA$. Definimos
\[
F=\{x\in A\mid \Box_p(x)=1\}\in A^\wedge,\quad a=\bigvee\{x\in A\mid \Diamond_p(x)=0\}
\]
\end{block}

\onslide<2->{\begin{block}{Teorema \citep{JohnstoneVietorisMonad}:}
Para cada marco $A$ los $V$-puntos son los pares $(F, a)$ donde 
\[
F\in A^\wedge\quad \text{y}\quad a\in [v_F(0), w_F(0)].
\]
\end{block}}

\onslide<3->{\begin{block}{Teorema \citep{sexton2006ordinal}:}
Para cada espacio sobrio $S$ los $V$-puntos de $\mathcal{O}S$ son los pares $(Q, X)$ donde
\[
\overline{M}\subseteq X\subseteq Q(\infty).
\]
\end{block}}
\end{frame}

\begin{frame}[fragile]
Sea $x\in A$ y $(A^\wedge, \mathcal{O}A^\wedge)$, donde 
\[
\mathcal{O}A^\wedge=\langle D(x)=\{F\in A^\wedge\mid x\in F\}\rangle  ,
\]
entonces
\[
\begin{split}
\pi\colon \pt(VA)\to A^\wedge \quad & \quad \mathcal{O}(\pi)=\pi^{-1}\colon \mathcal{O}A^\wedge \to \mathcal{O}\pt(VA) \\
(F, a)\mapsto F \quad & \quad D(x)\mapsto \mathcal{U}(\Box(x)).
\end{split}
\]

\begin{block}{Proposición:}
Sean $A\in \Frm$ y $A^\wedge$, $\pt(VA)$ como antes. Entonces las siguientes afirmaciones se cumplen:
\begin{enumerate}
  \item Si $A$ es $T_1$ y apilado, entonces $\mathcal{O}\pt(VA)\cong \mathcal{O}A^\wedge$.
  \item Si $\mathcal{O}\pt(VA)\cong \mathcal{O}A^\wedge$, entonces $A$ es apilado. 
  \item Si $\mathcal{O}\pt(VA)\cong \mathcal{O}A^\wedge$ y $A$ es espacial, entonces $A$ es $T_1$.
\end{enumerate}
\end{block}
\end{frame}

\begin{frame}
\begin{proof}
\begin{enumerate}
\item $T_1+$apilado $\Rightarrow$ $v_F(0)=w_F(0)$ para cada $F\in A^\wedge$. Por lo tanto, $\pi$ es una biyección y $\pi^{-1}$ es un morfismo de marcos biyectivo.
\item Si $\mathcal{O}\pt(VA)\cong \mathcal{O}A^\wedge$, entonces $\pi$ es una biyección. Por lo tanto, $v_F(0)=w_F(0)$ para cada $F\in A^\wedge$ y $A$ es apilado.
\item Sean $a\in A$ y $p\in \pt A$ tales que $p\leq a$. Si  $\mathcal{O}\pt(VA)\cong \mathcal{O}A^\wedge$, entonces $v_F(0)=w_F(0)$.
Por la espacialidad, 
\[
a=\bigwedge\{q\in \pt A\mid a\leq q\}\leq q=p.
\]
\end{enumerate}
\end{proof}
\end{frame}
